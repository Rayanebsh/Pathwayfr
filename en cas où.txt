    {% if token %}
    <script>
        const TOKEN = "{{ token | safe }}";  // injecté depuis Flask

        const form = document.getElementById("resetForm");
        const errorDiv = document.getElementById("error");

        form.addEventListener("submit", async function(e) {
            e.preventDefault();
            errorDiv.innerText = "";

            const password = document.getElementById("newPassword").value;
            const confirm = document.getElementById("confirmPassword").value;

            if (password !== confirm) {
                errorDiv.innerText = "Les mots de passe ne correspondent pas.";
                return;
            }

            try {
                const res = await fetch(`/auth/forgot-password/${TOKEN}`, {
                    method: "POST",
                    headers: {
                        "Content-Type": "application/json"
                    },
                    body: JSON.stringify({ password })
                });

                const data = await res.json();

                if (!res.ok) {
                    errorDiv.innerText = data.error || "Erreur inconnue.";
                } else {
                    window.location.href = "/auth/password-reset-success";
                }
            } catch (err) {
                errorDiv.innerText = "Erreur réseau.";
            }
        });
    </script>
    {% endif %}
